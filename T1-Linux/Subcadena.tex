\begin{frame}
  \frametitle{Subcadenas}
      %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
      Sean $x, y, z, w \in \Sigma^*$ palabras
      \begin{defi}
			$w$ es \textbf{subcadena} de $z$ si existen las cadenas $x$ e $y$ para las cuales $z = xwy$
      \end{defi}
      \pause

      \begin{block}{Ejemplo}
           \begin{itemize}[<+->]
           \item Sea $w = abc$. Subcadenas de $w$ son:
					 \item $\epsilon$
					 \item $a$
					 \item $b$
					 \item $c$
					 \item $ab$
					 \item $bc$
					 \item $abc$
					 \item $\ldots$
           \end{itemize}
			\end{block}
\end{frame}
