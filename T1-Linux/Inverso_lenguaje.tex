\begin{frame}
  \frametitle{Operaciones con Lenguajes}
      %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
      Sea $L \subseteq \Sigma^*$
      \begin{defi}
			\textbf{Inverso} de $L$: $L^{-1} = \{w \in \Sigma^*| \; w^i \in L \}$
      \end{defi}

			\pause
      \begin{block}{Ejemplo: si $L=\{alfa, beta\}$}
			$L^{-1} = \{ afla, ateb \}$
      \end{block}
\end{frame}
