					 \item $(A^+)^+ = A^+$
					 \item $(A^*)^* = A^*$
           \end{itemize}

           \begin{itemize}[<+->]
					 \item $(L^+)^* = L^*$
					 \item $(L^*)^+ = L^*$
					 \item $L^+ = L^* L = L L^*$
					 \item $L \subseteq L^+ \subseteq L^*$
					 \item $L_1 \subseteq L_2 \Rightarrow L_1^n \subseteq L_2^n \; \; \forall n \in \mathbb{N}$
					 \item $L_1 \subseteq L_2 \Rightarrow L_1^* \subseteq L_2^* \; \; (L_1^+ \subseteq L_2^+)$
           \end{itemize}
      \end{teor}
