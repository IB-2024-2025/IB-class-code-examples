\begin{frame}
  \frametitle{Operaciones con lenguajes}
      %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
			  Sea $L$ un lenguaje sobre un alfabeto $\Sigma$

      \pause
      \begin{defi}
			\textbf{Cierre de Kleene} (o cierre estrella) de $L$:

			$L ^* = \displaystyle\bigcup_{n=0}^{\infty} L^n$
      \end{defi}

			\pause
			Las cadenas de $L^*$ se forman realizando 0 칩 m치s concatenaciones de cadenas de $L$

      \pause
      \begin{defi}
			\textbf{Cierre positivo} de $L$:

			$L ^+ = \displaystyle\bigcup_{n=1}^{\infty} L^n$
			\end{defi}

			 Las cadenas de $L^+$ se forman realizando 1 칩 m치s concatenaciones de cadenas de $L$


\end{frame}
