  \frametitle{Operaciones con lenguajes}
      %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
			Sean $L_1, L_2, L_3$ lenguajes sobre $\Sigma$
      \begin{teor}
           \begin{itemize}[<+->]
           \item $L_1(L_2 \cup L_3) = L_1 L_2 \cup L_1 L_3$
           \item $(L_2 \cup L_3) L_1 = L_2 L_1 \cup L_3 L_1$
           \end{itemize}
      \end{teor}
